  -d '{
    "url": "https://docs.example.com",
    "maxPages": 50,

    }
  }'

import { Spidercrawl } from "spidercrawl";

const client = new Spidercrawl({ apiKey: "sk-sc-..." });

const job = await client.crawl({
  url: "https://docs.example.com",
  maxPages: 50,
  goal: "Extract all API endpoints",
  extractionSchema: {
    endpoint: "string",
    method: "string",
    description: "string",
  },
});

console.log(job.id); // job_abc123
from spidercrawl import Spidercrawl

client = Spidercrawl(api_key="sk-sc-...")

job = client.crawl(
    url="https://docs.example.com",
    max_pages=50,
    goal="Extract all API endpoints",
    extraction_schema={
        "endpoint": "string",
        "method": "string",
        "description": "string",
    },
)

print(job.id)  # job_abc123
